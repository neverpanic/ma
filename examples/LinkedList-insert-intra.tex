\def\hd{6em}%

\begin{tikzpicture}[font=\tiny\ttfamily]
	\path
		  (   0,    0)
		 +(-\hd,    0) node[cgarn,method] (list)       {list}
		 +(-\hd, -\vd) node[cgfrn,method] (head)       {head}

		 +(   0, -\vd) node[cgfrn,method] (next)       {next}
	;
